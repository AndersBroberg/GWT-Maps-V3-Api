001    package com.google.gwt.maps.client.services;
002    
003    /**
004     * Describes the type of location returned from a geocode.
005     * <br><br>
006     * See <a href="https://developers.google.com/maps/documentation/javascript/reference#GeocoderLocationType">GeocoderLocationType API Doc</a>
007     */
008    public enum GeocoderLocationType {
009    
010      /**
011       * The returned result is approximate.
012       */
013      APPROXIMATE,
014      
015      /**
016       * The returned result is the geometric center of a result such a line (e.g. street) or polygon (region).
017       */
018      GEOMETRIC_CENTER,
019      
020      /**
021       * The returned result reflects an approximation (usually on a road) interpolated between two precise points (such as intersections). Interpolated results are generally returned when rooftop geocodes are unavilable for a street address. 
022       */
023      RANGE_INTERPOLATED,
024      
025      /**
026       * The returned result reflects a precise geocode.
027       */
028      ROOFTOP;
029      
030      public String value() {
031        return name().toLowerCase();
032      }
033      
034      public static GeocoderLocationType fromValue(String type) {
035        return valueOf(type.toUpperCase());
036      }
037    
038      public String toString() {
039        return name();
040      }
041    }
