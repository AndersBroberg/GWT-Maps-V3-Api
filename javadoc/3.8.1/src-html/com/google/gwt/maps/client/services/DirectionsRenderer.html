001package com.google.gwt.maps.client.services;
002
003import com.google.gwt.core.client.JavaScriptObject;
004import com.google.gwt.dom.client.Element;
005import com.google.gwt.event.shared.HandlerRegistration;
006import com.google.gwt.maps.client.MapImpl;
007import com.google.gwt.maps.client.MapWidget;
008import com.google.gwt.maps.client.events.MapEventType;
009import com.google.gwt.maps.client.events.MapHandlerRegistration;
010import com.google.gwt.maps.client.events.directions.DirectionsChangeEventFormatter;
011import com.google.gwt.maps.client.events.directions.DirectionsChangeMapHandler;
012import com.google.gwt.maps.client.mvc.MVCObject;
013
014/**
015 * Renders directions retrieved in the form of a DirectionsResult object
016 * retrieved from the DirectionsService. This class extends MVCObject. <br>
017 * <br>
018 * See <a href=
019 * "https://developers.google.com/maps/documentation/javascript/reference#DirectionsRenderer"
020 * >DirectionsRenderer API Doc</a>
021 */
022public class DirectionsRenderer extends MVCObject<DirectionsRenderer> {
023
024        /**
025         * use newInstance();
026         */
027        protected DirectionsRenderer() {
028        }
029
030        /**
031         * Renders directions retrieved in the form of a DirectionsResult object
032         * retrieved from the DirectionsService. This class extends MVCObject.
033         * 
034         * @param options
035         */
036        public static final DirectionsRenderer newInstance(
037                        DirectionsRendererOptions options) {
038                JavaScriptObject jso = createJso(options);
039                return jso.cast();
040        }
041
042        private static final native JavaScriptObject createJso(
043                        DirectionsRendererOptions options) /*-{
044                return new $wnd.google.maps.DirectionsRenderer(options);
045        }-*/;
046
047        /**
048         * Returns the renderer's current set of directions.
049         */
050        public final native DirectionsResult getDirections() /*-{
051                return this.getDirections();
052        }-*/;
053
054        /**
055         * This method specifies the map on which directions will be rendered. Pass
056         * null to remove the directions from the map.
057         * 
058         * @param mapWidget
059         */
060        public final void setMap(MapWidget mapWidget) {
061                if (mapWidget == null) {
062                        setMapImpl(null);
063                } else {
064                        setMapImpl(mapWidget.getJso());
065                }
066        }
067
068        private final native void setMapImpl(MapImpl map) /*-{
069                this.setMap(map);
070        }-*/;
071
072        /**
073         * Returns the map on which the DirectionsResult is rendered.
074         */
075        public final MapWidget getMap() {
076                return MapWidget.newInstance(getMapImpl());
077        }
078
079        private final native MapImpl getMapImpl() /*-{
080                return this.getMap();
081        }-*/;
082
083        /**
084         * Returns the panel <code>&lt;div&gt;</code> in which the DirectionsResult
085         * is rendered.
086         */
087        public final native Element getPanel() /*-{
088                return this.getPanel();
089        }-*/;
090
091        /**
092         * Returns the current (zero-based) route index in use by this
093         * DirectionsRenderer object.
094         */
095        public final native int getRouteIndex() /*-{
096                return this.getRouteIndex();
097        }-*/;
098
099        /**
100         * Set the renderer to use the result from the DirectionsService. Setting a
101         * valid set of directions in this manner will display the directions on the
102         * renderer's designated map and panel.
103         * 
104         * @param directions
105         */
106        public final native void setDirections(DirectionsResult directions) /*-{
107                this.setDirections(directions);
108        }-*/;
109
110        /**
111         * Change the options settings of this DirectionsRenderer after
112         * initialization.
113         * 
114         * @param options
115         */
116        public final native void setOptions(DirectionsRendererOptions options) /*-{
117                this.setOptions(options);
118        }-*/;
119
120        /**
121         * This method renders the directions in a <code>&lt;div&gt;</code>. Pass
122         * null to remove the content from the panel.
123         * 
124         * @param element
125         */
126        public final native void setPanel(Element element) /*-{
127                this.setPanel(element);
128        }-*/;
129
130        /**
131         * Set the (zero-based) index of the route in the DirectionsResult object to
132         * render. By default, the first route in the array will be rendered.
133         * 
134         * @param routeIndex
135         */
136        public final native void setRouteIndex(int routeIndex) /*-{
137                this.setRouteIndex(routeIndex);
138        }-*/;
139
140        /**
141         * This event is fired when the rendered directions change, either when a
142         * new DirectionsResult is set or when the user finishes dragging a change
143         * to the directions path.
144         * 
145         * @param handler
146         */
147        public final HandlerRegistration addDirectionsChangeHandler(
148                        DirectionsChangeMapHandler handler) {
149                return MapHandlerRegistration.addHandler(this,
150                                MapEventType.DIRECTIONS_CHANGED, handler,
151                                new DirectionsChangeEventFormatter());
152        }
153
154}
