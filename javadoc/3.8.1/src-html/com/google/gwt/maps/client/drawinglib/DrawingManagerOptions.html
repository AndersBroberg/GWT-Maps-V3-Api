001package com.google.gwt.maps.client.drawinglib;
002
003import com.google.gwt.core.client.JavaScriptObject;
004import com.google.gwt.maps.client.MapImpl;
005import com.google.gwt.maps.client.MapWidget;
006import com.google.gwt.maps.client.overlays.CircleOptions;
007import com.google.gwt.maps.client.overlays.MarkerOptions;
008import com.google.gwt.maps.client.overlays.PolygonOptions;
009import com.google.gwt.maps.client.overlays.PolylineOptions;
010import com.google.gwt.maps.client.overlays.RectangleOptions;
011
012/**
013 * Options for the drawing manager. <br>
014 * <br>
015 * See <a href=
016 * "https://developers.google.com/maps/documentation/javascript/reference#DrawingManagerOptions"
017 * >DrawingManagerOptions API Doc</a>
018 */
019public class DrawingManagerOptions extends JavaScriptObject {
020
021        /**
022         * use newInstance();
023         */
024        protected DrawingManagerOptions() {
025        }
026
027        /**
028         * Options for the drawing manager.
029         */
030        public static final DrawingManagerOptions newInstance() {
031                DrawingManagerOptions obj = JavaScriptObject.createObject().cast();
032                return setDefaults(obj);
033        }
034        
035        private static DrawingManagerOptions setDefaults(DrawingManagerOptions obj) {
036                obj.setDrawingControl(true);
037                return obj;
038        }
039
040        /**
041         * Options to apply to any new circles created with this DrawingManager. The
042         * center and radius properties are ignored, and the map property of a new
043         * circle is always set to the DrawingManager's map.
044         * 
045         * @param circleOptions
046         */
047        public final native void setCircleOptions(CircleOptions circleOptions) /*-{
048                this.circleOptions = circleOptions;
049        }-*/;
050
051        /**
052         * Options to apply to any new circles created with this DrawingManager. The
053         * center and radius properties are ignored, and the map property of a new
054         * circle is always set to the DrawingManager's map.
055         */
056        public final native CircleOptions getCircleOptions() /*-{
057                return this.circleOptions;
058        }-*/;
059
060        /**
061         * The enabled/disabled state of the drawing control. Defaults to true.
062         * 
063         * @param drawingControl
064         */
065        public final native void setDrawingControl(boolean drawingControl) /*-{
066                this.drawingControl = drawingControl;
067        }-*/;
068
069        /**
070         * The enabled/disabled state of the drawing control. Defaults to true.
071         */
072        public final native boolean getDrawingControl() /*-{
073                return this.drawingControl;
074        }-*/;
075
076        /**
077         * The display options for the drawing control.
078         * 
079         * @param drawingControlOptions
080         */
081        public final native void setDrawingControlOptions(
082                        DrawingControlOptions drawingControlOptions) /*-{
083                this.drawingControlOptions = drawingControlOptions;
084        }-*/;
085
086        /**
087         * The display options for the drawing control.
088         */
089        public final native DrawingControlOptions getDrawingControlOptions() /*-{
090                return this.drawingControlOptions;
091        }-*/;
092
093        /**
094         * The DrawingManager's drawing mode, which defines the type of overlay to
095         * be added on the map. Accepted values are MARKER, POLYGON, POLYLINE,
096         * RECTANGLE, CIRCLE, or null. A drawing mode of null means that the user
097         * can interact with the map as normal, and clicks do not draw anything.
098         * 
099         * @param drawingMode
100         */
101        public final void setDrawingMode(OverlayType drawingMode) {
102                setDrawingModeImpl(drawingMode.value());
103        }
104
105        /**
106         * The DrawingManager's drawing mode, which defines the type of overlay to
107         * be added on the map. Accepted values are MARKER, POLYGON, POLYLINE,
108         * RECTANGLE, CIRCLE, or null. A drawing mode of null means that the user
109         * can interact with the map as normal, and clicks do not draw anything.
110         * 
111         * @param drawingMode
112         */
113        private native void setDrawingModeImpl(String drawingMode) /*-{
114                this.drawingMode = drawingMode;
115        }-*/;
116
117        /**
118         * The Map to which the DrawingManager is attached, which is the Map on
119         * which the overlays created will be placed.
120         */
121        public final MapWidget getMap() {
122                return MapWidget.newInstance(getMapImpl());
123        }
124
125        private final native MapImpl getMapImpl() /*-{
126                return this.map;
127        }-*/;
128
129        /**
130         * The Map to which the DrawingManager is attached, which is the Map on
131         * which the overlays created will be placed.
132         * 
133         * @param mapWidget
134         */
135        public final void setMap(MapWidget mapWidget) {
136                MapImpl value = (mapWidget != null) ? mapWidget.getJso() : null;
137                setMapImpl(value);
138        }
139
140        private final native void setMapImpl(MapImpl map) /*-{
141                this.map = map;
142        }-*/;
143
144        /**
145         * Options to apply to any new markers created with this DrawingManager. The
146         * position property is ignored, and the map property of a new marker is
147         * always set to the DrawingManager's map.
148         * 
149         * @param markerOptions
150         */
151        public final native void setMarkerOptions(MarkerOptions markerOptions) /*-{
152                this.markerOptions = markerOptions;
153        }-*/;
154
155        /**
156         * Options to apply to any new markers created with this DrawingManager. The
157         * position property is ignored, and the map property of a new marker is
158         * always set to the DrawingManager's map.
159         */
160        public final native MarkerOptions getMarkerOptions() /*-{
161                return this.markerOptions;
162        }-*/;
163
164        /**
165         * Options to apply to any new polygons created with this DrawingManager.
166         * The paths property is ignored, and the map property of a new polygon is
167         * always set to the DrawingManager's map.
168         * 
169         * @param polygonOptions
170         */
171        public final native void setPolygonOptions(PolygonOptions polygonOptions) /*-{
172                this.polygonOptions = polygonOptions;
173        }-*/;
174
175        /**
176         * Options to apply to any new polygons created with this DrawingManager.
177         * The paths property is ignored, and the map property of a new polygon is
178         * always set to the DrawingManager's map.
179         */
180        public final native PolygonOptions getPolygonOptions() /*-{
181                return this.polygonOptions;
182        }-*/;
183
184        /**
185         * Options to apply to any new polylines created with this DrawingManager.
186         * The path property is ignored, and the map property of a new polyline is
187         * always set to the DrawingManager's map.
188         * 
189         * @param polylineOptions
190         */
191        public final native void setPolylineOptions(PolylineOptions polylineOptions) /*-{
192                this.polylineOptions = polylineOptions;
193        }-*/;
194
195        /**
196         * Options to apply to any new polylines created with this DrawingManager.
197         * The path property is ignored, and the map property of a new polyline is
198         * always set to the DrawingManager's map.
199         */
200        public final native PolylineOptions getPolylineOptions() /*-{
201                return this.polylineOptions;
202        }-*/;
203
204        /**
205         * Options to apply to any new rectangles created with this DrawingManager.
206         * The bounds property is ignored, and the map property of a new rectangle
207         * is always set to the DrawingManager's map.
208         * 
209         * @param rectangleOptions
210         */
211        public final native void setRectangleOptions(
212                        RectangleOptions rectangleOptions) /*-{
213                this.rectangleOptions = rectangleOptions;
214        }-*/;
215
216        /**
217         * Options to apply to any new rectangles created with this DrawingManager.
218         * The bounds property is ignored, and the map property of a new rectangle
219         * is always set to the DrawingManager's map.
220         */
221        public final native RectangleOptions getRectangleOptions() /*-{
222                return this.rectangleOptions;
223        }-*/;
224
225}
