001package com.google.gwt.maps.client.drawinglib;
002
003import com.google.gwt.core.client.JavaScriptObject;
004import com.google.gwt.event.shared.HandlerRegistration;
005import com.google.gwt.maps.client.MapImpl;
006import com.google.gwt.maps.client.MapWidget;
007import com.google.gwt.maps.client.events.MapEventType;
008import com.google.gwt.maps.client.events.MapHandlerRegistration;
009import com.google.gwt.maps.client.events.overlaycomplete.OverlayCompleteEventFormatter;
010import com.google.gwt.maps.client.events.overlaycomplete.OverlayCompleteMapHandler;
011import com.google.gwt.maps.client.events.overlaycomplete.circle.CircleCompleteEventFormatter;
012import com.google.gwt.maps.client.events.overlaycomplete.circle.CircleCompleteMapHandler;
013import com.google.gwt.maps.client.events.overlaycomplete.marker.MarkerCompleteEventFormatter;
014import com.google.gwt.maps.client.events.overlaycomplete.marker.MarkerCompleteMapHandler;
015import com.google.gwt.maps.client.events.overlaycomplete.polygon.PolygonCompleteEventFormatter;
016import com.google.gwt.maps.client.events.overlaycomplete.polygon.PolygonCompleteMapHandler;
017import com.google.gwt.maps.client.events.overlaycomplete.polyline.PolylineCompleteEventFormatter;
018import com.google.gwt.maps.client.events.overlaycomplete.polyline.PolylineCompleteMapHandler;
019import com.google.gwt.maps.client.events.overlaycomplete.rectangle.RectangleCompleteEventFormatter;
020import com.google.gwt.maps.client.events.overlaycomplete.rectangle.RectangleCompleteMapHandler;
021import com.google.gwt.maps.client.mvc.MVCObject;
022
023/**
024 * Allows users to draw markers, polygons, polylines, rectangles, and circles on
025 * the map. The DrawingManager's drawing mode defines the type of overlay that
026 * will be created by the user. Adds a control to the map, allowing the user to
027 * switch drawing mode. This class extends MVCObject. <br>
028 * <br>
029 * See <a href=
030 * "https://developers.google.com/maps/documentation/javascript/reference#DrawingManager"
031 * >DrawingManager API Doc</a>
032 */
033public class DrawingManager extends MVCObject<DrawingManager> {
034
035        /**
036         * use newInstance();
037         */
038        protected DrawingManager() {
039        }
040
041        /**
042         * Creates a DrawingManager that allows users to draw overlays on the map,
043         * and switch between the type of overlay to be drawn with a drawing
044         * control.
045         * 
046         * @param options
047         */
048        public static final DrawingManager newInstance(DrawingManagerOptions options) {
049                return createJso(options).cast();
050        }
051
052        private static native JavaScriptObject createJso(
053                        DrawingManagerOptions options) /*-{
054                return new $wnd.google.maps.drawing.DrawingManager(options);
055        }-*/;
056
057        /**
058         * Returns the DrawingManager's drawing mode.
059         */
060        public final OverlayType getDrawingMode() {
061                String type = getDrawingModeImpl();
062                if(type==null) {
063                        return null;
064                }
065                else {
066                        return OverlayType.fromValue(type);                     
067                }
068        }
069
070        private final native String getDrawingModeImpl() /*-{
071                return this.getDrawingMode();
072        }-*/;
073
074        /**
075         * Changes the DrawingManager's drawing mode, which defines the type of
076         * overlay to be added on the map. Accepted values are MARKER, POLYGON,
077         * POLYLINE, RECTANGLE, CIRCLE, or null. A drawing mode of null means that
078         * the user can interact with the map as normal, and clicks do not draw
079         * anything.
080         * 
081         * @param drawingMode
082         */
083        public final void setDrawingMode(OverlayType drawingMode) {
084                String value = (drawingMode!=null) ? drawingMode.value() : null;
085                setDrawingModeImpl(value);
086        }
087
088        private final native void setDrawingModeImpl(String drawingMode) /*-{
089                this.setDrawingMode(drawingMode);
090        }-*/;
091
092        /**
093         * Attaches the DrawingManager object to the specified Map.
094         * 
095         * @param mapWidget
096         */
097        public final void setMap(MapWidget mapWidget) {
098                MapImpl value = (mapWidget!=null) ? mapWidget.getJso() : null;
099                setMapImpl(value);
100        }
101
102        private final native void setMapImpl(MapImpl map) /*-{
103                this.setMap(map);
104        }-*/;
105
106        /**
107         * Returns the Map to which the DrawingManager is attached, which is the Map
108         * on which the overlays created will be placed.
109         */
110        public final MapWidget getMap() {
111                return MapWidget.newInstance(getMapImpl());
112        }
113
114        private final native MapImpl getMapImpl() /*-{
115                return this.getMap();
116        }-*/;
117
118        /**
119         * This event is fired when the user has finished drawing a circle.
120         * 
121         * @param handler
122         */
123        public final HandlerRegistration addCircleCompleteHandler(
124                        CircleCompleteMapHandler handler) {
125                return MapHandlerRegistration.addHandlerDrawing(this,
126                                MapEventType.CIRCLECOMPLETE, handler,
127                                new CircleCompleteEventFormatter());
128        }
129
130        /**
131         * This event is fired when the user has finished drawing a marker.
132         * 
133         * @param handler
134         */
135        public final HandlerRegistration addMarkerCompleteHandler(
136                        MarkerCompleteMapHandler handler) {
137                return MapHandlerRegistration.addHandlerDrawing(this,
138                                MapEventType.MARKERCOMPLETE, handler,
139                                new MarkerCompleteEventFormatter());
140        }
141
142        /**
143         * This event is fired when the user has finished drawing an overlay of any
144         * type.
145         * 
146         * @param handler
147         */
148        public final HandlerRegistration addOverlayCompleteHandler(
149                        OverlayCompleteMapHandler handler) {
150                return MapHandlerRegistration.addHandler(this,
151                                MapEventType.OVERLAYCOMPLETE, handler,
152                                new OverlayCompleteEventFormatter());
153        }
154
155        /**
156         * This event is fired when the user has finished drawing a polygon.
157         * 
158         * @param handler
159         */
160        public final HandlerRegistration addPolygonCompleteHandler(
161                        PolygonCompleteMapHandler handler) {
162                return MapHandlerRegistration.addHandlerDrawing(this,
163                                MapEventType.POLYGONCOMPLETE, handler,
164                                new PolygonCompleteEventFormatter());
165        }
166
167        /**
168         * This event is fired when the user has finished drawing a polyline.
169         * 
170         * @param handler
171         */
172        public final HandlerRegistration addPolylineCompleteHandler(
173                        PolylineCompleteMapHandler handler) {
174                return MapHandlerRegistration.addHandlerDrawing(this,
175                                MapEventType.POLYLINECOMPLETE, handler,
176                                new PolylineCompleteEventFormatter());
177        }
178
179        /**
180         * This event is fired when the user has finished drawing a rectangle.
181         * 
182         * @param handler
183         */
184        public final HandlerRegistration addRectangleCompleteHandler(
185                        RectangleCompleteMapHandler handler) {
186                return MapHandlerRegistration.addHandlerDrawing(this,
187                                MapEventType.RECTANGLECOMPLETE, handler,
188                                new RectangleCompleteEventFormatter());
189        }
190
191}
