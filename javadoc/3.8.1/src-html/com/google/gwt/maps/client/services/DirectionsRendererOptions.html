001package com.google.gwt.maps.client.services;

003import com.google.gwt.core.client.JavaScriptObject;
004import com.google.gwt.dom.client.Element;
005import com.google.gwt.maps.client.MapImpl;
006import com.google.gwt.maps.client.MapWidget;
007import com.google.gwt.maps.client.overlays.InfoWindow;

011/**
012 * This object defines the properties that can be set on a DirectionsRenderer
013 * object. <br>
014 * <br>
015 * See <a href=
016 * "https://developers.google.com/maps/documentation/javascript/reference#DirectionsRendererOptions"
017 * >DirectionsRendererOptions API Doc</a>
018 */
019public class DirectionsRendererOptions extends JavaScriptObject {
020
021        /**
022         * use newInstance();
023         */
024        protected DirectionsRendererOptions() {
025        }
026
027        /**
028         * This object defines the properties that can be set on a
029         * DirectionsRenderer object.
030         */
031        public static final DirectionsRendererOptions newInstance() {
032                return JavaScriptObject.createObject().cast();
033        }
034
035        /**
036         * The directions to display on the map and/or in a <code>&lt;div&gt;</code>
037         * panel, retrieved as a DirectionsResult object from DirectionsService.
038         * 
039         * @param directions
040         */
041        public final native void setDirections(DirectionsResult directions) /*-{
042                this.directions = directions;
043        }-*/;
044
045        /**
046         * The directions to display on the map and/or in a <code>&lt;div&gt;</code>
047         * panel, retrieved as a DirectionsResult object from DirectionsService.
048         */
049        public final native DirectionsResult getDirections() /*-{
050                return this.directions;
051        }-*/;
052
053        /**
054         * If true, allows the user to drag and modify the paths of routes rendered
055         * by this DirectionsRenderer.
056         * 
057         * @param draggable
058         */
059        public final native void setDraggable(boolean draggable) /*-{
060                this.draggable = draggable;
061        }-*/;
062
063        /**
064         * If true, allows the user to drag and modify the paths of routes rendered
065         * by this DirectionsRenderer.
066         */
067        public final native boolean getDraggable() /*-{
068                return this.draggable;
069        }-*/;
070
071        /**
072         * This property indicates whether the renderer should provide UI to select
073         * amongst alternative routes. By default, this flag is false and a
074         * user-selectable list of routes will be shown in the directions'
075         * associated panel. To hide that list, set hideRouteList to true.
076         * 
077         * @param hideRouteList
078         */
079        public final native void setHideRouteList(boolean hideRouteList) /*-{
080                this.hideRouteList = hideRouteList;
081        }-*/;
082
083        /**
084         * This property indicates whether the renderer should provide UI to select
085         * amongst alternative routes. By default, this flag is false and a
086         * user-selectable list of routes will be shown in the directions'
087         * associated panel. To hide that list, set hideRouteList to true.
088         */
089        public final native boolean getHideRouteList() /*-{
090                return this.hideRouteList;
091        }-*/;
092
093        /**
094         * The InfoWindow in which to render text information when a marker is
095         * clicked. Existing info window content will be overwritten and its
096         * position moved. If no info window is specified, the DirectionsRenderer
097         * will create and use its own info window. This property will be ignored if
098         * suppressInfoWindows is set to true.
099         * 
100         * @param infoWindow
101         */
102        public final native void setInfoWindow(InfoWindow infoWindow) /*-{
103                this.infoWindow = infoWindow;
104        }-*/;
105
106        /**
107         * The InfoWindow in which to render text information when a marker is
108         * clicked. Existing info window content will be overwritten and its
109         * position moved. If no info window is specified, the DirectionsRenderer
110         * will create and use its own info window. This property will be ignored if
111         * suppressInfoWindows is set to true.
112         */
113        public final native InfoWindow getInfoWindow() /*-{
114                return this.infoWindow;
115        }-*/;
116
117        /**
118         * Map on which to display the directions.
119         * 
120         * @param mapWidget
121         */
122        public final void setMap(MapWidget mapWidget) {
123                if (mapWidget == null) {
124                        setMapImpl(null);
125                } else {
126                        setMapImpl(mapWidget.getJso());
127                }
128        }
129
130        private final native void setMapImpl(MapImpl map) /*-{
131                this.map = map;
132        }-*/;
133
134        /**
135         * Map on which to display the directions.
136         */
137        public final MapWidget getMap() {
138                return MapWidget.newInstance(getMapImpl());
139        }
140
141        /**
142         * Map on which to display the directions.
143         * 
144         * @deprecated See {@link #getMap()}
145         */
146        @Deprecated
147        public final MapWidget getMapWidget() {
148                return MapWidget.newInstance(getMapImpl());
149        }
150
151        private final native MapImpl getMapImpl() /*-{
152                return this.map;
153        }-*/;
154
155        /**
156         * Options for the markers. All markers rendered by the DirectionsRenderer
157         * will use these options.
158         * 
159         * @param markerOptions
160         */
161        public final native void setMarkerOptions(MarkerOptions markerOptions) /*-{
162                this.markerOptions = markerOptions;
163        }-*/;
164
165        /**
166         * Options for the markers. All markers rendered by the DirectionsRenderer
167         * will use these options.
168         */
169        public final native MarkerOptions getMarkerOptions() /*-{
170                return this.markerOptions;
171        }-*/;
172
173        /**
174         * The <code>&lt;div&gt;</code> in which to display the directions steps.
175         * 
176         * @param panel
177         */
178        public final native void setPanel(Element panel) /*-{
179                this.panel = panel;
180        }-*/;
181
182        /**
183         * The <code>&lt;div&gt;</code> in which to display the directions steps.
184         */
185        public final native Element getPanel() /*-{
186                return this.panel;
187        }-*/;
188
189        /**
190         * Options for the polylines. All polylines rendered by the
191         * DirectionsRenderer will use these options.
192         * 
193         * @param polylineOptions
194         */
195        public final native void setPolylineOptions(PolylineOptions polylineOptions) /*-{
196                this.polylineOptions = polylineOptions;
197        }-*/;
198
199        /**
200         * Options for the polylines. All polylines rendered by the
201         * DirectionsRenderer will use these options.
202         */
203        public final native PolylineOptions getPolylineOptions() /*-{
204                return this.polylineOptions;
205        }-*/;
206
207        /**
208         * By default, the input map is centered and zoomed to the bounding box of
209         * this set of directions. If this option is set to true, the viewport is
210         * left unchanged, unless the map's center and zoom were never set.
211         * 
212         * @param preserveViewport
213         */
214        public final native void setPreserveViewport(boolean preserveViewport) /*-{
215                this.preserveViewport = preserveViewport;
216        }-*/;
217
218        /**
219         * By default, the input map is centered and zoomed to the bounding box of
220         * this set of directions. If this option is set to true, the viewport is
221         * left unchanged, unless the map's center and zoom were never set.
222         */
223        public final native boolean getPreserveViewport() /*-{
224                return this.preserveViewport;
225        }-*/;

227        /**
228         * The index of the route within the DirectionsResult object. The default

234                this.routeIndex = routeIndex;
235        }-*/;
236
237        /**
238         * The index of the route within the DirectionsResult object. The default

241        public final native int getRouteIndex() /*-{
242                return this.routeIndex;
243        }-*/;

261                return this.suppressBicyclingLayer;

270                this.suppressInfoWindows = suppressInfoWindows;

308        public final native boolean getSuppressPolylines() /*-{
