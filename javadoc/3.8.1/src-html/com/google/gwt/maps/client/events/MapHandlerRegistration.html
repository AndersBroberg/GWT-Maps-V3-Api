016         * event handler
017         * 
018         * @param jso
019         * @param eventType
020         * @param handler
021         * @param formatter
022         */
023        @SuppressWarnings("rawtypes")
024        // is ugly, but is a cyclic generic type, so suppressed
025        public static <E extends MapEvent> HandlerRegistration addHandler(
026                        JavaScriptObject jso, MapEventType eventType,
027                        MapHandler<E> handler, MapEventFormatter<E> formatter) {
028                final JavaScriptObject listener = addHandlerImpl(jso,
029                                eventType.value(), handler, formatter);
030                HandlerRegistration registration = new HandlerRegistration() {
031                        @Override
032                        public void removeHandler() {
033                                removeHandlerImpl(listener);
034                        }
035                };
036                return registration;
037        }

160        /**
161         * process generic handler callback
162         * 
163         * @param handler
164         * @param properties
165         * @param formatter
166         */
167        @SuppressWarnings("rawtypes")
168        // is ugly, but is a cyclic generic type, so suppressed
169        protected static <E extends MapEvent> void onCallback(
170                        final MapHandler<E> handler, final Properties properties,
171                        final MapEventFormatter<E> formatter) {
172                try {
173                        formatEvent(handler, properties, formatter);
174                } catch (Throwable x) {
175                        GWT.getUncaughtExceptionHandler().onUncaughtException(x);
176                }
177        }
178
179        /**
180         * create a map event
181         * 
182         * @param handler
183         * @param properties
184         * @param formatter
185         */
186        @SuppressWarnings("rawtypes")
187        // is ugly, but is a cyclic generic type, so suppressed
188        private static <E extends MapEvent> void formatEvent(MapHandler<E> handler,
189                        Properties properties, MapEventFormatter<E> formatter) {
190                E event = formatter.createEvent(properties);
191                handler.onEvent(event);
192        }

194        /**
195         * Triggers the given event. All arguments after eventName are passed as
196         * arguments to the listeners. <br>
197         * <b>Note:</b> Use Marker in objects to click on.
198         * 
199         * @param w
200         * @param eventType
201         * @param objects
202         */
203        public static void trigger(MapWidget w, MapEventType eventType,
204                        JavaScriptObject... objects) {
205                JsArray<JavaScriptObject> a = ArrayHelper.toJsArray(objects);
206                triggerImpl(w, eventType.value(), a);
207        }

249        }

251        private static native <E> void addDomListener(JavaScriptObject object,
252                        String eventName, MapHandler<?> handler, boolean capture) /*-{
253                var callback = function(event) {
254                        @com.google.gwt.maps.client.events.MapHandlerRegistration::addDomListenerImpl(Lcom/google/gwt/ajaxloader/client/Properties;Lcom/google/gwt/maps/client/events/MapHandler;)(event, handler)
255                };
256                $wnd.google.maps.event.addDomListener(object, eventName, callback,
257                                capture);
258        }-*/;
259
260        @SuppressWarnings("rawtypes")
261        // is ugly, but is a cyclic generic type, so suppressed
262        private static void addDomListenerImpl(Properties properties,
263                        MapHandler<MapEvent> handler) {
264                MapChangeEventFormatter formatter = new MapChangeEventFormatter();
265                handler.onEvent(formatter.createEvent(properties));
266        }
