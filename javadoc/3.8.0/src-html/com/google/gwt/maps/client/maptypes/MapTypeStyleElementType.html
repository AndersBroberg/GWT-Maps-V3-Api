001    package com.google.gwt.maps.client.maptypes;
002    
003    /**
004     * Each MapTypeStyleElementType distinguishes between the different representations of a feature.
005     * <br><br>
006     * See <a href="https://developers.google.com/maps/documentation/javascript/reference#MapTypeStyleElementType">MapTypeStyleElementType API Doc</a>
007     */
008    public enum MapTypeStyleElementType {
009    
010      /**
011       * Apply the rule to all elements of the specified feature.
012       */
013      ALL,
014      
015      /**
016       * Apply the rule to the feature's geometry.
017       */
018      GEOMETRY,
019      
020      /**
021       * Apply the rule to the feature's labels.
022       */
023      LABELS;
024      
025      /**
026       * return the enum value as a String
027       * @return String
028       */
029      public String value() {
030        return name().toLowerCase();
031      }
032    
033      /**
034       * convert a String value to enum Type
035       * @param type
036       * @return TYPE
037       */
038      public static MapTypeStyleElementType fromValue(String type) {
039        return valueOf(type.toUpperCase());
040      }
041    
042      /**
043       * return the enum Type as a String
044       */
045      @Override
046      public String toString() {
047        return name().toLowerCase();
048      }
049    }
