001    package com.google.gwt.maps.client.maptypes;
002    
003    import com.google.gwt.dom.client.Document;
004    import com.google.gwt.dom.client.Element;
005    import com.google.gwt.maps.client.base.Point;
006    import com.google.gwt.maps.client.base.Size;
007    
008    /**
009     * This interface defines map type. This interface is typically used for base maps such as road, satellite or hybrid maps. Immutable.
010     * <br><br>
011     * See <a href="https://developers.google.com/maps/documentation/javascript/reference#MapType">MapType API Doc</a>
012     */
013    public interface MapType {
014      
015      /**
016       * Returns a tile for the given tile coordinate (x, y) and zoom level. This tile will be appended to the given ownerDocument.
017       * @param tileCoord
018       * @param zoom
019       * @param ownerDocument
020       */
021      public Element getTile(Point tileCoord, int zoom, Document ownerDocument);
022      
023      /**
024       * Releases the given tile, performing any necessary cleanup. The provided tile will have already been removed from the document. Optional.
025       * @param tile
026       */
027      public void releaseTile(Element tile);
028      
029      /**
030       * sets Alt text to display when this MapType's button is hovered over in the MapTypeControl. Optional.
031       * @param alt
032       */
033      public void setAlt(String alt);
034      
035      /**
036       * gets Alt text to display when this MapType's button is hovered over in the MapTypeControl. Optional.
037       */
038      public String getAlt();
039      
040      /**
041       * sets The maximum zoom level for the map when displaying this MapType. Required for base MapTypes, ignored for overlay MapTypes.
042       * @param maxZoom
043       */
044      public void setMaxZoom(int maxZoom);
045      
046      /**
047       * sets The minimum zoom level for the map when displaying this MapType. Optional; defaults to 0.
048       * @param minZoom
049       */
050      public void setMinZoom(int minZoom);
051      
052      /**
053       * sets Name to display in the MapTypeControl. Optional.
054       * @param name
055       */
056      public void setName(String name);
057      
058      /**
059       * sets The Projection used to render this MapType. Optional; defaults to Mercator.
060       * @param projection
061       */
062      public void setProjection(Projection projection);
063      
064      /**
065       * sets Radius of the planet for the map, in meters. Optional; defaults to Earth's equatorial radius of 6378137 meters.
066       * @param radius
067       */
068      public void setRadius(int radius);
069      
070      /**
071       * sets The dimensions of each tile. Required.
072       * @param tileSize
073       */
074      public void setTileSize(Size tileSize);
075    }
